\begin{definition}[Block geometry]
Fix a 4D lattice torus $(\mathbb{Z}/L\mathbb{Z})^4$ with spacing $a$.
Let $\mathcal{B}_j$ denote the partition into disjoint hypercubes of side $2^j a$.
Define the block-collapsing map $\pi_j$ from fields to block-averaged fields.
\end{definition}
\begin{definition}[Local algebra]
Let $\mathfrak{A}(\Omega)$ be the $*$-algebra generated by plaquette observables supported in $\Omega$.
Define the radius-$R$ neighborhood $\Omega^{(R)}$ in graph distance.
\end{definition}
