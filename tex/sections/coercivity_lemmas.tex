\section{Coercivity Lemma Chain}

\begin{lemma}[Local Gauge PoincarÃ©]
Let $A$ be gauge-fixed on a block $B$. Then
\[
\langle \psi, P_B \psi\rangle \ge c_0 \|\psi\|^2
\quad \forall \psi \perp \mathcal Z_B .
\]
\end{lemma}

\begin{lemma}[IMS Localization]
For any partition $\{\chi_i\}$,
\[
P = \sum_i \chi_i P \chi_i - \sum_i |\nabla \chi_i|^2 ,
\]
with error uniformly bounded.
\end{lemma}

\begin{lemma}[Mourre Estimate]
There exists a conjugate operator $A_B$ such that
\[
[P_B, iA_B] \ge c_1 > 0
\]
on $\mathcal Z_B^\perp$.
\end{lemma}

\begin{lemma}[Uniform Block Coercivity]
There exists $\kappa>0$ independent of scale such that
\[
P_B \ge \kappa \quad \text{on } \mathcal Z_B^\perp .
\]
\end{lemma}
