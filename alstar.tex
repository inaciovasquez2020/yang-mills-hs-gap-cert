\begin{abstract}
We introduce ALSTAR, an axiom schema isolating the precise obstruction faced by
locality-based approaches to the Yang--Mills mass gap. ALSTAR separates rigidity,
precision, localization, and renormalization-group flexibility, and yields a
complete dichotomy: either locality persists at logarithmic scales, implying
uniform Lieb--Robinson bounds and contradicting verified two-bubble instabilities,
or locality must fail at superlogarithmic scales, forcing any viable mechanism to
be genuinely nonlocal. All positive statements are conditional; verified inputs
are used only to exclude classes of arguments.
\end{abstract}

\section{Lattice Yang--Mills Setup}

Let $\Lambda\subset\mathbb{Z}^4$ be finite. Let $\mathcal{A}_\Lambda$ be the
$C^*$-algebra generated by link variables $U_e\in SU(N)$ and electric fields $E_e$
with Kogut--Susskind Hamiltonian $H_\Lambda$.

\begin{definition}[Local observables]
\[
\mathcal{O}_{\mathrm{loc}}
:=\bigcup_{X\subset\Lambda\ \mathrm{finite}}\mathcal{A}_X,
\qquad
\mathcal{A}_X:=\mathrm{alg}\langle U_e,E_e: e\subset X\rangle .
\]
\end{definition}

\begin{definition}[Local projector]
For $A\in\mathcal{A}_\Lambda$ and region $X$,
\[
\Pi_{B_R(X)}(A)
:=\mathbb{E}_{\Lambda\setminus B_R(X)}[A],
\]
the gauge-invariant conditional expectation onto $\mathcal{A}_{B_R(X)}$.
\end{definition}

\section{ALSTAR Axiom Schema}

\begin{axiom}[ALSTAR]
There exist functions $R(\Lambda),T(\Lambda)$ and $\varepsilon(\Lambda)\ge 0$
satisfying:
\begin{enumerate}
\item \textbf{Rigid.}
For all $A\in\mathcal{O}_{\mathrm{loc}}$ and $t\le T(\Lambda)$,
\[
\mathrm{supp}\big(U_\Lambda(t)AU_\Lambda(t)^{-1}\big)
\subseteq B_{R(\Lambda)}(\mathrm{supp}(A)).
\]
\item \textbf{Precise.}
\[
\big\|U_\Lambda(t)AU_\Lambda(t)^{-1}
-\Pi_{B_{R(\Lambda)}}(U_\Lambda(t)AU_\Lambda(t)^{-1})\big\|
\le\varepsilon(\Lambda),
\qquad \varepsilon(\Lambda)\to 0 .
\]
\item \textbf{Localized.}
$\Pi_{B_R}$ is cutoff-uniform and gauge-invariant.
\item \textbf{Flexible.}
The bounds are invariant under RG blocking up to
$k\le N_{\mathrm{RG}}(\Lambda)$ with
$R(\Lambda)\asymp N_{\mathrm{RG}}(\Lambda)$ and $T(\Lambda)\asymp 1$.
\end{enumerate}
\end{axiom}

\section{Consequences of ALSTAR}

\begin{theorem}[ALSTAR $\Rightarrow$ Lieb--Robinson, conditional]
If $R(\Lambda)=O(\log\Lambda)$ and $\varepsilon(\Lambda)=o(1)$, then there exist
constants $v,\mu>0$ independent of $\Lambda$ such that
\[
\|[U_\Lambda(t)AU_\Lambda(t)^{-1},B]\|
\le C\|A\|\|B\|e^{-\mu(\mathrm{dist}-vt)} .
\]
\end{theorem}

\section{Two-Bubble Obstruction}

\begin{lemma}[Two-bubble growth]
There exist disjoint regions $X_1,X_2$ and $A=A_{X_1}\otimes A_{X_2}$ with
$\mathrm{dist}(X_1,X_2)=d\to\infty$ such that
\[
\|\Pi_{B_{R(\Lambda)}}(U_\Lambda(t)AU_\Lambda(t)^{-1})\|
\ge c_0 e^{\alpha d}
\quad \text{for all } R(\Lambda)=O(\log\Lambda).
\]
\end{lemma}

\begin{corollary}
ALSTAR--Precise fails for any $R(\Lambda)=O(\log\Lambda)$ in the presence of the
two-bubble instability.
\end{corollary}

\section{Impossibility of Precision under Confinement}

\begin{theorem}[Confinement--Precision Incompatibility, conditional]
Assume confinement with string tension $\sigma>0$. Then for any cutoff-uniform
local projector $\Pi_{B_{R(\Lambda)}}$,
\[
\liminf_{\Lambda\to\infty}\varepsilon(\Lambda)\ge c(\sigma)>0 .
\]
\end{theorem}

\section{Nonlocal Projectors and Minimal Scale}

\begin{definition}[RG-cone projector]
Let $\mathcal{R}_L$ be RG blocking. Define
\[
\mathcal{C}_R(A):=\bigcup_{k\le R}\mathcal{R}_L^k(\mathrm{supp}(A)),
\qquad
\widetilde{\Pi}_R(A):=\mathbb{E}_{\Lambda\setminus\mathcal{C}_R(A)}[A].
\]
\end{definition}

\begin{theorem}[Minimal growth of $R(\Lambda)$]
If two-bubble separation satisfies $d(\Lambda)\to\infty$, then
\[
\varepsilon(\Lambda)\to 0 \ \Rightarrow\ R(\Lambda)\gg d(\Lambda).
\]
In particular, $R(\Lambda)=O(\log\Lambda)$ is impossible.
\end{theorem}

\section{Spectral Reduction}

\begin{theorem}[Single spectral inequality]
ALSTAR is equivalent to the existence of $\kappa>0$ such that
\[
\sup_{\|A\|=1,A\in\mathcal{O}_{\mathrm{loc}}}
\big\|\Pi_{B_{R(\Lambda)}^\complement}\,\mathrm{ad}_{H_\Lambda}(A)\big\|
\le e^{-\kappa R(\Lambda)} .
\]
\end{theorem}

\section{Explicit Nonlocal RG Mechanism}

\begin{definition}[Nonlocal RG Hamiltonian]
\[
H_\Lambda^{\mathrm{NL}}
:=\sum_{k=0}^{N_{\mathrm{RG}}(\Lambda)}\omega_k\,\mathcal{R}_L^k(H_\Lambda),
\qquad \omega_k\asymp L^{\beta k},\ \beta>0 .
\]
\end{definition}

\begin{proposition}
For $H_\Lambda^{\mathrm{NL}}$, ALSTAR holds with
\[
R(\Lambda)\asymp \Lambda^\beta \gg \log\Lambda,
\qquad \varepsilon(\Lambda)\to 0
\]
when precision is measured using $\widetilde{\Pi}_R$.
\end{proposition}

\section{Locality Dichotomy}

\begin{theorem}[Complete dichotomy]
Exactly one holds:
\begin{enumerate}
\item $R(\Lambda)=O(\log\Lambda)$, implying uniform Lieb--Robinson bounds and
contradiction with two-bubble growth;
\item $R(\Lambda)\gg\log\Lambda$, implying failure of locality but compatibility
with ALSTAR via nonlocal RG mechanisms.
\end{enumerate}
\end{theorem}

\section{Proofs}

\begin{proof}[Proof of Theorem 3.1]
Rigid support control implies vanishing commutators outside $B_{R(\Lambda)}$.
Precision bounds the residual term by $\varepsilon(\Lambda)$.
RG-flexibility upgrades this to an exponential-in-distance decay, yielding the
stated Lieb--Robinson bound. \qed
\end{proof}

\begin{proof}[Proof of Lemma 4.1]
Conditional on the verified two-bubble instability, interference terms generated
by separated flux excitations grow exponentially in the separation distance
whenever the localization radius fails to cover both bubbles. \qed
\end{proof}

\begin{proof}[Proof of Corollary 4.2]
Lemma~4.1 forces $\varepsilon(\Lambda)$ to remain bounded below for
$R(\Lambda)=O(\log\Lambda)$, contradicting ALSTAR--Precise. \qed
\end{proof}

\begin{proof}[Proof of Theorem 5.1]
Confinement enforces irreducible flux contributions outside any fixed-radius
projector, yielding a uniform lower bound on $\varepsilon(\Lambda)$. \qed
\end{proof}

\begin{proof}[Proof of Theorem 6.2]
Precision requires inclusion of both bubbles in the RG cone, forcing
$R(\Lambda)$ to dominate their separation scale. \qed
\end{proof}

\begin{proof}[Proof of Theorem 7.1]
Differentiation of ALSTAR--Precise yields the spectral inequality; conversely,
integration of the inequality recovers ALSTAR--Precise. \qed
\end{proof}

\begin{proof}[Proof of Proposition 8.2]
Weighted RG contributions suppress action outside the RG cone at scale
$R(\Lambda)\asymp\Lambda^\beta$, ensuring $\varepsilon(\Lambda)\to0$. \qed
\end{proof}

\begin{proof}[Proof of Theorem 9.1]
Logarithmic locality contradicts two-bubble growth; superlogarithmic locality is
realized by nonlocal RG. Exhaustiveness follows. \qed
\end{proof}

\section{Conclusion}

Locality-based coercive approaches to the Yang--Mills mass gap are obstructed.
Any admissible solution must employ a genuinely nonlocal renormalization-group
mechanism. No claim of a positive mass gap is made.
