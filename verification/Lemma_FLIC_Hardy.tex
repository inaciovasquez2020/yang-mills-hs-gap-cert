\begin{lemma}[FLIC implies Hardy inequality]
Assume there exists \(C_{\mathrm{FLIC}}\) such that for all instantons \(A\) on \(S^4\),
\[
\sup_{x\in S^4,\, r>0} r^{-2}\!\int_{B(x,r)} |F_A|^2 \le C_{\mathrm{FLIC}}.
\]
Then for all \(\phi\in H^1(S^4)\),
\[
\int_{S^4} |F_A|^2|\phi|^2
\le \tfrac14\int_{S^4}|\nabla\phi|^2 + C\int_{S^4}|\phi|^2,
\]
with \(C\) depending only on \(C_{\mathrm{FLIC}}\) and the geometry of \(S^4\).
\end{lemma}
